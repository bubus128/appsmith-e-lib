SELECT
    count(*)
FROM
    public.users
JOIN
    public.audiobookrents ON public.users.user_id = public.audiobookrents.userid
JOIN
    public.audiobooks ON public.audiobookrents.audiobookid = public.audiobooks.id
JOIN
    public.books ON public.audiobooks.bookid = public.books.id
WHERE
    public.users.username = '{{appsmith.store.user}}';
