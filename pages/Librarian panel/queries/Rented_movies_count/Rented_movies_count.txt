SELECT
    count(*)
FROM
    public.users
JOIN
    public.moviesrent ON public.users.user_id = public.moviesrent.userid
JOIN
    public.moviecopies ON public.moviesrent.moviecopyid = public.moviecopies.id
JOIN
    public.movies ON public.moviecopies.movieid = public.movies.id
WHERE
    public.users.username = '{{appsmith.store.user}}';
